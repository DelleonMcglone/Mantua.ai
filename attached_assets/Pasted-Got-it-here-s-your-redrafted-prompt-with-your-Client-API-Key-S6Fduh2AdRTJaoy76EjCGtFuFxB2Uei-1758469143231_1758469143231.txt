
3. **Create Providers File**
   Make a new file called `providers.tsx` and set up the OnchainKitProvider:

   ```tsx
   'use client';

   import type { ReactNode } from 'react';
   import { OnchainKitProvider } from '@coinbase/onchainkit';
   import { base } from 'wagmi/chains'; // use baseSepolia for testing

   export function Providers(props: { children: ReactNode }) {
     return (
       <OnchainKitProvider
         apiKey={process.env.NEXT_PUBLIC_ONCHAINKIT_API_KEY}
         chain={base} // swap for baseSepolia in testing
       >
         {props.children}
       </OnchainKitProvider>
     );
   }
   ```

4. **Wrap the App with `<Providers />`**
   In your `app.tsx` or `layout.tsx`, wrap your app with the `Providers` component:

   ```tsx
   import './globals.css';
   import { Providers } from './providers'; 

   export default function RootLayout({ children }: { children: React.ReactNode }) {
     return (
       <html lang="en">
         <body>
           <Providers>
             {children}
           </Providers>
         </body>
       </html>
     );
   }
   ```

5. **Add OnchainKit Styles**
   Import the styles so the components are pre-styled:

   ```tsx
   import '@coinbase/onchainkit/styles.css';
   import './globals.css';
   ```

   ```tsx
   import '@coinbase/onchainkit/styles.css';
   ```
