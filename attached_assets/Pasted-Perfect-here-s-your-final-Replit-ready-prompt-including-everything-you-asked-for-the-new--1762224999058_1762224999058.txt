```tsx
/**
 * Mantua Add Liquidity Component (Final)
 * --------------------------------------
 * - Full Range / Custom Range functionality
 * - Mantua Intel Hook in dropdown
 * - Set Price Range positioned below Hook Selector
 * - Dark Mantua theme
 */

import React, { useState } from "react";
import { Button } from "@/components/ui/button";

const hooks = [
  "No Hook",
  "Mantua Intel Hook",
  "Dynamic Fee Hook",
  "TWAMM Hook",
  "MEV Protection Hook",
];

const AddLiquidity: React.FC = () => {
  const [token1, setToken1] = useState("");
  const [token2, setToken2] = useState("");
  const [selectedHook, setSelectedHook] = useState("No Hook");
  const [rangeType, setRangeType] = useState<"full" | "custom">("full");
  const [minPrice, setMinPrice] = useState("");
  const [maxPrice, setMaxPrice] = useState("");

  return (
    <div className="bg-black text-white rounded-2xl p-6 max-w-xl mx-auto space-y-6">
      <h2 className="text-xl font-semibold">Add liquidity to a pool.</h2>

      {/* --- Token 1 --- */}
      <div>
        <label className="block mb-1 text-sm text-zinc-400">Token 1</label>
        <div className="flex gap-2">
          <input
            type="number"
            placeholder="0.0"
            className="w-full bg-zinc-900 p-3 rounded-lg"
            value={token1}
            onChange={(e) => setToken1(e.target.value)}
          />
          <button className="bg-zinc-800 px-3 rounded-lg text-zinc-300">
            Select
          </button>
          <button className="bg-zinc-800 px-3 rounded-lg text-zinc-300">
            Max
          </button>
        </div>
      </div>

      {/* --- Token 2 --- */}
      <div>
        <label className="block mb-1 text-sm text-zinc-400">Token 2</label>
        <div className="flex gap-2">
          <input
            type="number"
            placeholder="0.0"
            className="w-full bg-zinc-900 p-3 rounded-lg"
            value={token2}
            onChange={(e) => setToken2(e.target.value)}
          />
          <button className="bg-zinc-800 px-3 rounded-lg text-zinc-300">
            Select
          </button>
          <button className="bg-zinc-800 px-3 rounded-lg text-zinc-300">
            Max
          </button>
        </div>
      </div>

      {/* --- Fee Tier --- */}
      <div>
        <label className="block text-sm mb-2 text-zinc-400">Fee tiers</label>
        <select className="w-full bg-zinc-900 p-3 rounded-lg">
          <option>0.30% ‚Äì most common</option>
          <option>0.05%</option>
          <option>0.10%</option>
        </select>
      </div>

      {/* --- Hook Selector --- */}
      <div>
        <label className="block text-sm mb-2 text-zinc-400">
          Select a hook (optional)
        </label>
        <select
          value={selectedHook}
          onChange={(e) => setSelectedHook(e.target.value)}
          className="w-full bg-zinc-900 p-3 rounded-lg"
        >
          {hooks.map((hook) => (
            <option key={hook} value={hook}>
              {hook}
            </option>
          ))}
        </select>
      </div>

      {/* --- Set Price Range Section --- */}
      <div className="bg-zinc-900 p-5 rounded-xl space-y-3">
        <h3 className="font-semibold text-lg">Set price range</h3>

        {/* Range Toggle */}
        <div className="flex gap-2">
          <button
            onClick={() => setRangeType("full")}
            className={`flex-1 py-2 rounded-lg ${
              rangeType === "full"
                ? "bg-pink-600 text-white"
                : "bg-zinc-800 text-zinc-300"
            }`}
          >
            Full range
          </button>
          <button
            onClick={() => setRangeType("custom")}
            className={`flex-1 py-2 rounded-lg ${
              rangeType === "custom"
                ? "bg-pink-600 text-white"
                : "bg-zinc-800 text-zinc-300"
            }`}
          >
            Custom range
          </button>
        </div>

        {/* Description */}
        {rangeType === "full" ? (
          <p className="text-zinc-400 text-sm">
            Providing full range liquidity ensures continuous market
            participation across all possible prices, offering simplicity but
            with potential for higher impermanent loss.
          </p>
        ) : (
          <p className="text-zinc-400 text-sm">
            Custom range allows you to concentrate your liquidity within
            specific price bounds, enhancing capital efficiency and fee
            earnings but requiring more active management.
          </p>
        )}

        {/* --- Current Price --- */}
        <div className="bg-zinc-800 p-3 rounded-lg flex justify-between items-center">
          <div>
            <p className="text-sm text-zinc-400">Current price</p>
            <h4 className="font-bold text-lg">3,638.71</h4>
            <p className="text-sm text-zinc-400">ETH/USDC ($3,638.71)</p>
          </div>
          <div className="flex gap-2">
            <div className="bg-zinc-700 px-3 py-1 rounded-lg text-sm">ETH</div>
            <div className="bg-zinc-700 px-3 py-1 rounded-lg text-sm">USDC</div>
          </div>
        </div>

        {/* --- Chart Placeholder --- */}
        <div className="bg-gradient-to-t from-purple-900 via-zinc-900 to-black rounded-lg mt-4 h-40 relative">
          <p className="absolute inset-0 flex items-center justify-center text-zinc-500 text-sm">
            [ Price Range Chart Visualization Placeholder ]
          </p>
        </div>

        {/* --- Custom Range Inputs --- */}
        {rangeType === "custom" && (
          <div className="flex gap-2 mt-4">
            <input
              type="number"
              placeholder="Min price"
              value={minPrice}
              onChange={(e) => setMinPrice(e.target.value)}
              className="w-1/2 bg-zinc-800 p-3 rounded-lg"
            />
            <input
              type="number"
              placeholder="Max price"
              value={maxPrice}
              onChange={(e) => setMaxPrice(e.target.value)}
              className="w-1/2 bg-zinc-800 p-3 rounded-lg"
            />
          </div>
        )}

        {/* --- Time Range Buttons + Reset --- */}
        <div className="flex justify-between items-center text-sm text-zinc-400 mt-2">
          <div className="flex gap-3">
            <button>1D</button>
            <button>1W</button>
            <button className="text-pink-400">1M</button>
            <button>1Y</button>
            <button>All time</button>
          </div>
          <button className="text-pink-500">Reset</button>
        </div>
      </div>

      {/* --- Add Liquidity Button --- */}
      <Button className="w-full bg-gradient-to-r from-pink-500 to-purple-600 text-white text-lg py-3 rounded-xl">
        Add liquidity
      </Button>
    </div>
  );
};

export default AddLiquidity;
```
